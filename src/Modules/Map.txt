class Pair (
	def Pair {first, second} (this.first = first; this.second = second)
	def toString = "{" ++ show first ++ ", " ++ show second ++ "}"
)

class Map (
	def Map {} (this.pairs = [])
	
	def get key = (
		for p in pairs (
			if (p.first == key) (return p.second)
		)
		return ?
	)

	def put {key, value} = (
		for p in pairs (
			if (p.first == key) (
				prev = p.second
				p.second = value
				return prev
			)
		)
		this.pairs = Pair {key, value} : pairs
		return ?
	)

	def keys = map (.first) pairs
	
	def values = map (.second) pairs
)
